on createSnapshot
   get the short name of the current stack
   put it into lStack
   put the number of the current card into lCard
   add - 1 to lCard
   lock screen
   put (specialFolderPath("resources")&"/Documents"&"/Page"& lCard &"_"& lStack & ".png") into tPathFile
   export snapshot from rect "0,0,1368,764" to file tPathFile as PNG
   unlock screen
   answer "Votre fichier a bien t enregistr dans le dossier Documents"
end createSnapshot

              put (gCurrentTotal-tCurrentValue) into gCurrentTotal

      put (gZoomLevel*widthField) into widthField
      put (gZoomLevel*heightField) into heightField
      put (gZoomLevel*xField) into xField
      put (gZoomLevel*yField) into yField

      set the width of fld id fldID to widthField
      set the height of fld id fldID to heightField

   else if gZoomSelected = "-" then
      put (gZoomLevel*widthField) into widthField
      put (gZoomLevel*heightField) into heightField
      put (gZoomLevel*xField) into xField
      put (gZoomLevel*yField) into yField
      set the loc of fld id fldID to xField,yField

      set the height of fld id fldID to heightField

   end if
end zoomField
